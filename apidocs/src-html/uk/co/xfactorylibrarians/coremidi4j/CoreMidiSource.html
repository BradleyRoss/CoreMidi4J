082  /**
083   * Opens the Core MIDI Device
084   *
085   * @throws MidiUnavailableException if the MIDI system cannot be used
086   * 
087   */
088
089  @Override
090  public void open() throws MidiUnavailableException {
091
092    try {
093
094      // Create the input port if not already created
095      if (this.input == null) {
096
097        this.input = CoreMidiDeviceProvider.getMIDIClient().inputPortCreate("Core Midi Provider Input");
098
099      }
100
101      // And connect to it
102      this.input.connectSource(this);
103      isOpen = true;
104      
105      // Get the system time in microseconds
106      startTime = this.getMicroSecondTime();
107
108    } catch (CoreMidiException e) {
109
110      e.printStackTrace();
111      throw new MidiUnavailableException(e.getMessage());
112
113    }
114
115  }

117  /**
118   * Closes the Core MIDI Device
119   * 
120   */
121
122  @Override
123  public void close() {
124
125    try {
126
127      // If the port is created then disconnect from it
128      if (this.input != null) {
129
130        this.input.disconnectSource(this);
131
132      }
133
134      // Clear the transmitter list
135      synchronized (transmitters) {
136
137        transmitters.clear();
138
139      }
140
141    } catch (CoreMidiException e) {
142
143      e.printStackTrace();
144
145    } finally {
146
147      // Reset the context data
148      isOpen = false;
149
150    }
151
152  }

309  /**
310   * Checks whether a status byte represents a real-time message, which can occur even in the middle of another
311   * multi-byte message.
312   *
313   * @param status the status byte which has just been received.
314   *
315   * @return true if the status byte represents a standalone real-time message which should be passed on without
316   *         interrupting any other message being gathered.
317   */
318  private boolean isRealTimeMessage(byte status) {
319
320    switch ( status ) {
321
322      case (byte) ShortMessage.TIMING_CLOCK:
323      case (byte) ShortMessage.START:
324      case (byte) ShortMessage.CONTINUE:
325      case (byte) ShortMessage.STOP:
326      case (byte) ShortMessage.ACTIVE_SENSING:
327      case (byte) ShortMessage.SYSTEM_RESET:
328        return  true;
329
330      default:
331        return false;
332
333    }
334
335  }
336
337  /**
338   * Checks whether a status byte represents a running-status message, which means that multiple messages can be
339   * sent without re-sending the status byte, for example to support multiple note-on messages in a row by simply
340   * sending a stream of data byte pairs after the note-on status byte.
341   *
342   * @param status the status byte which is being processed.
343   *
344   * @return true if we should stay in this status after receiving our full complement of data bytes.
345   */
346  private boolean isRunningStatusMessage (int status) {
347
348    switch( status & 0xF0 ) {
349
350      case ShortMessage.NOTE_OFF:
351      case ShortMessage.NOTE_ON: 
352      case ShortMessage.POLY_PRESSURE:
353      case ShortMessage.CONTROL_CHANGE:
354      case ShortMessage.PROGRAM_CHANGE: 
355      case ShortMessage.CHANNEL_PRESSURE:
356      case ShortMessage.PITCH_BEND:
357        return true;
358
359      default:
360        return false;
361
362    }
363
364  }
365
366  /**
367   * Determine how many data bytes are expected for a given MIDI message other than a SYSEX message, which varies.
368   *
369   * @param status the status byte introducing the MIDI message.
370   *
371   * @return the number of data bytes which must be received for the message to be complete.
372   *
373   * @throws InvalidMidiDataException if the status byte is not valid.
374   */
375  private int expectedDataLength (byte status) throws InvalidMidiDataException {
376
377    // system common and system real-time messages
378
379    switch( status &0xFF ) {
380
381      case ShortMessage.TUNE_REQUEST:
382      case ShortMessage.END_OF_EXCLUSIVE:
383
384        // System real-time messages
385      case ShortMessage.TIMING_CLOCK:  
386      case 0xF9:  // Undefined
387      case ShortMessage.START:  
388      case ShortMessage.CONTINUE:  
389      case ShortMessage.STOP:  
390      case 0xFD:  // Undefined
391      case ShortMessage.ACTIVE_SENSING:  
392      case ShortMessage.SYSTEM_RESET:  
393        return 0;
394
395      case ShortMessage.MIDI_TIME_CODE:
396      case ShortMessage.SONG_SELECT:  
397        return 1;
398
399      case ShortMessage.SONG_POSITION_POINTER:  
400        return 2;
401
402      default:  // Fall through to next switch
403
404    }
405
406    // channel voice and mode messages
407    switch( status & 0xF0 ) {
408
409      case ShortMessage.NOTE_OFF: 
410      case ShortMessage.NOTE_ON:  
411      case ShortMessage.POLY_PRESSURE:
412      case ShortMessage.CONTROL_CHANGE:  
413      case ShortMessage.PITCH_BEND: 
414        return 2;
415
416      case ShortMessage.PROGRAM_CHANGE:  
417      case ShortMessage.CHANNEL_PRESSURE:  
418        return 1;
419
420      default:
421        throw new InvalidMidiDataException("Invalid status byte: " + status);
422
423    }
424
425  }

427  /**
428   * The message callback for receiving midi data from the JNI code
429   * 
430   * @param coreTimestamp  The time in microseconds since boot at which the messages should take effect
431   * @param packetlength   The length of the packet of messages
432   * @param data           The data array that holds the messages
433   * 
434   * @throws InvalidMidiDataException if the message contained values that could not be interpreted as valid MIDI
435   * 
436   */
437
438  public void messageCallback(long coreTimestamp, int packetlength, byte data[]) throws InvalidMidiDataException {
439
440    int offset = 0;
441
442    // Convert from CoreMIDI-oriented boot-relative microseconds to Java-oriented port-relative microsecends,
443    // and from unsigned CoreMIDI semantics of 0 meaning now to signed Java semantics of -1 meaning now.
444    final long timestamp = (coreTimestamp == 0) ? -1 : coreTimestamp - startTime;
445
446    // An OSX MIDI packet may contain multiple messages
447    while (offset < packetlength) {
448
449      if (data[offset] >= 0) {
450
451        // This is a regular data byte. Process it appropriately for our current message type.
452        if (currentMessage == 0) {
453
454          throw new InvalidMidiDataException("Data received outside of a message.");
455
456        } else if (currentMessage == SysexMessage.SYSTEM_EXCLUSIVE) {
457
458          // We are in the middle of gathering system exclusive data; continue.
459          offset += processSysexData(packetlength, data, offset, timestamp);
460
461        } else if (currentDataIsSingleByte) {
462
463          // We are processing a message which only needs one data byte, this completes it
464          transmitMessage(new ShortMessage(currentMessage, data[offset++], 0), timestamp);
465          if (!isRunningStatusMessage(currentMessage)) currentMessage = 0;
466
467        } else {
468
469          // We are processing a message which needs two data bytes
470          if (wasFirstByteReceived) {
471
472            // We have the second data byte, the message is now complete
473            transmitMessage(new ShortMessage(currentMessage, firstDataByte, data[offset++]), timestamp);
474            wasFirstByteReceived = false;
475            if (!isRunningStatusMessage(currentMessage)) currentMessage = 0;
476
477          } else {
478
479            // We have just received the first data byte of a message which needs two.
480            firstDataByte = data[offset++];
481            wasFirstByteReceived = true;
482            
483          }
484          
485        }
486
487      } else {
488
489        // This is a status byte, handle appropriately
490        if (isRealTimeMessage(data[offset])) {
491
492          // Real-time messages can come anywhere, including in between data bytes of other messages.
493          // Simply transmit it and move on.
494          transmitMessage(new ShortMessage(data[offset++] & 0xff), timestamp);
495
496        } else if (data[offset] == (byte) ShortMessage.END_OF_EXCLUSIVE) {
497
498          // This is the marker for the end of a system exclusive message. If we were gathering one,
499          // process (finish) it.
500          if (currentMessage == SysexMessage.SYSTEM_EXCLUSIVE) {
501
502            offset += processSysexData(packetlength, data, offset, timestamp);
503          } else {
504
505            throw new InvalidMidiDataException("Received End of Exclusive marker outside SYSEX message");
506
507          }
508
509        } else if (data[offset] == (byte) SysexMessage.SYSTEM_EXCLUSIVE) {
510
511          // We are starting to gather a SYSEX message.
512          currentMessage = SysexMessage.SYSTEM_EXCLUSIVE;
513          sysexMessageLength = 0;
514          sysexMessageData = new Vector<byte[]>();
515          offset += processSysexData(packetlength, data, offset, timestamp);
516
517        } else {
518
519          // Some ordinary MIDI message.
520          switch (expectedDataLength(data[offset])) {
521
522            case 0:  // No data bytes, this is a standalone message, so we can send it right away.
523              transmitMessage(new ShortMessage(data[offset++] & 0xff), timestamp);
524              currentMessage = 0;  // If we were in a running status, it's over now
525              break;
526
527            case 1: // We are expecting data before we can send this message.
528              currentMessage = data[offset++] & 0xff;
529              currentDataIsSingleByte = true;
530              break;
531
532            case 2: // We are expecting two bytes of data before we can send this message.
533              currentMessage = data[offset++] & 0xff;
534              currentDataIsSingleByte = false;
535              wasFirstByteReceived = false;
536              break;
537
538            default:
539              throw new InvalidMidiDataException("Unexpected data length: " +
540                  expectedDataLength(data[offset]));
541
542          }
543
544        }
545
546      }
547
548    }
549
550  }

552  /**
553   * Creates a SYSEX message from the received (potentially partial) messages. This function is called when F7 was
554   * detected in the most recent message gathered, indicating the end of the SYSEX.
555   *
556   * @return The constructed SYSEX message
557   * 
558   * @throws InvalidMidiDataException  Thrown if MIDI data malformed.
559   * 
560   */
561
562  private SysexMessage constructSysexMessage() throws InvalidMidiDataException {
563
564    // Create the array to hold the constructed message and reset the index (where the data will be copied)
565    byte data[] = new byte[sysexMessageLength];
566    int index = 0;
567
568    // Iterate through the partial messages
569    for (int i = 0; i < sysexMessageData.size(); i += 1) {
570
571      // Get the partial message
572      byte sourceData[] = sysexMessageData.get(i);
573
574      // Copy the partial message into the array
575      System.arraycopy(sourceData, 0, data, index, sourceData.length);
576
577      // Point the index to where the next partial message needs to be copied.
578      index += sourceData.length;
579
580    }
581
582    // We are done with the message fragments, so allow them to be garbage collected
583    sysexMessageData = null;
584
585    // Create and return the new SYSYEX Message
586    return new SysexMessage(data, sysexMessageLength);
587
588  }

590  /**
591   * Called when a SYSEX message is being received, either because an F0 byte has been seen at the start of a
592   * message, which starts the process of gathering a SYSEX potentially across multiple packets, or because a
593   * new packet has been received while we are still in the process of gathering bytes of a SYSEX which was
594   * started in a previous message. The partial data is added to the sysexMessageData Vector. If we see another
595   * status byte, except for one which represents a real-time message, we know the SYSEX is finished, and so we
596   * can assemble and transmit it from any fragments which have been gathered.
597   *
598   * If we see a real-time message or the end of the packet, we return the data we have gathered,
599   * and let the main loop decide what to do next.
600   *
601   * @param packetLength        The length of the data packet
602   * @param sourceData          The source data received from Core MIDI
603   * @param startOffset         The position within the packet where the current message began
604   * @param timestamp           The message timestamp
605   *
606   * @return                                                    The number of bytes consumed from the packet by the SYSEX message.
607   * 
608   * @throws     InvalidMidiDataException   Thrown if MIDI data malformed.
609   * 
610   */
611
612  private int processSysexData(int packetLength, byte sourceData[], int startOffset, long timestamp)
613      throws InvalidMidiDataException {
614
615    // Look for the end of the SYSEX or packet
616    int messageLength = 0;
617    boolean foundEnd = false;
618
619    // Check to see if this packet contains the end of the message
620    while ( ( startOffset + messageLength ) < packetLength) {
621
622      byte latest = sourceData[startOffset + messageLength++];
623
624      if (latest < 0 && messageLength + sysexMessageLength > 1) {
625
626        // We have encountered another status byte (after the F0 which starts the message).
627        // Is it the marker of the end of the SYSEX message?
628        if (latest == (byte) ShortMessage.END_OF_EXCLUSIVE) {
629
630          currentMessage = 0;  // Found end marker, ready to send this SYSEX and look for next message.
631          foundEnd = true;
632
633        } else if (isRealTimeMessage(latest)) {
634
635          // Back up so the main loop can send this real-time message embedded in our data,
636          // then call us again to keep gathering SYSEX data.
637          --messageLength;
638
639        } else {
640
641          // Found the start of another message. Back up so it gets processed, and note that we have found
642          // the end of our SYSEX data. If we decide we should not pass on incomplete SYSEX messages, the
643          // code below the loop can use the fact that currentMessage is not 0 to do so.
644          --messageLength;
645          foundEnd = true;
646
647        }
648
649        break;  // One way or another, we are done gathering data bytes for now.
650
651      }
652
653    }
654
655    // Create an array to hold this part of the message, if we received any actual data.
656    // (Note the source array will be released by the native function.)
657    if (messageLength > 0) {
658
659      byte data[] = new byte[messageLength];
660
661      //Copy the data to the array
662      try {
663
664        System.arraycopy(sourceData, startOffset, data, 0, messageLength);
665
666      } catch (ArrayIndexOutOfBoundsException e) {
667
668        e.printStackTrace();
669
670        throw e;
671
672      }
673
674      // Add the message to the vector
675      sysexMessageData.add(data);
676
677    }
678
679    // Update the length of the SYSEX message
680    sysexMessageLength += messageLength;
681
682    // If we found the end, send it now
683    if (foundEnd) {
684
685      // Again, here we could refrain from sending if currentMessage != 0, because that indicates we received
686      // a partial SYSEX message, i.e. the next message started before we received the End of Exclusive marker.
687      transmitMessage(constructSysexMessage(), timestamp);
688
689    }
690
691    return messageLength;
692
693  }

696  /**
697   * Sends a MIDI message to all of the registered transmitters
698   *
699   * @param message             the message to send
700   * @param timestamp   the time stamp
701   * 
702   */
703
704  private void transmitMessage(final MidiMessage message, long timestamp) {
705
706    // Uncomment the following to filter realtime messages during debugging
707    //          if (isRealTimeMessage ((byte)message.getStatus())) {
708    //
709    //                  return;
710    //
711    //          }
712
713    synchronized (transmitters) {
714
715      // Get the iterators from the transmitters collection
716      final Iterator<Transmitter> iterator = transmitters.iterator();
717
718      // Loop through the transmitters
719      while (iterator.hasNext()) {
720
721        // Get the next transmitter
722        final Transmitter transmitter = iterator.next();
723
724        // If the transmitter is not null then get the receiver
725        if (transmitter != null) {
726
727          final Receiver receiver = transmitter.getReceiver();
728
729          // If the receiver is not null then get send the message
730          if (receiver != null) {
731
732            receiver.send(message, timestamp);
733
734          }
735
736        }
737
738      }
739
740    }
741
742  }

744  /**
745   * Formats the provided data into a HEX string, which is useful for debugging
746   *
747   * @param aByte The data to format
748   * 
749   * @return The formatted HEX string
750   * 
751   */
752
753  private String getHexString(byte[] aByte) {
754
755    StringBuffer sbuf = new StringBuffer(aByte.length * 3 + 2);
756
757    for (int i = 0; i < aByte.length; i++) {
758
759      sbuf.append(' ');
760      byte bhigh = (byte) ((aByte[i] & 0xf0) >> 4);
761      sbuf.append((char) (bhigh > 9 ? bhigh + 'A' - 10 : bhigh + '0'));
762      byte blow = (byte) (aByte[i] & 0x0f);
763      sbuf.append((char) (blow > 9 ? blow + 'A' - 10 : blow + '0'));
764
765    }
766
767    return new String(sbuf).trim();
768
769  }
